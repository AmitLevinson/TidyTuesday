clean_avg <- cran_code %>% 
  add_count(language) %>% 
  filter(n > 20) %>% 
  group_by(language) %>% 
  summarise(count = n(),
    avg = mean(code)) %>% 
  mutate(language = fct_reorder(language, avg)) %>% 
  arrange(desc(avg))

  geom_point(aes(size = count), color = "red")+
